   MIN(SAL) 최저연봉 FROM EMP GROUP BY JOB;
   
-- 그룹에 대한 조건을 줄경우 HAVING 
--위의 SQL문에 인원수가 3명이상인 그룹만 출력
SELECT JOB 직업,COUNT(*) 인원수,ROUND(AVG(SAL),0) 평균연봉,MAX(SAL) 최고연봉,
   MIN(SAL) 최저연봉 FROM EMP GROUP BY JOB HAVING COUNT(*)>=3;
