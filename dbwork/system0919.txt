--scott 계정을 잠금상태로 변경해보자~ 
--alter 는 수정이라고 이해하면 편함
alter user scott account lock;

--scott 계정 락을 풀어보자
alter user scott account UNLOCK;

--engel 계정을 만들어 보자 (비번 a1234)
create user angel IDENTIFIED BY a1234;
create user c##angel IDENTIFIED BY a1234;

--생성된 계정을 삭제하려면
drop user c##angel;

--c## 을 안붙이고 계정을 생성하려면 세션을 변경해야만 한다.
alter session set "_ORACLE_SCRIPT"=true;
create user angel IDENTIFIED BY a1234;

grant connect,resource to angel;

--angel 에 테이블을 생성할 수 있도록 tablespace 잡기
